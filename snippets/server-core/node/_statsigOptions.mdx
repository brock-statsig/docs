```javascript
import { Statsig, StatsigOptions } from '@statsig/statsig-node-core';

const options: StatsigOptions = {
  environment: "production",    // Environment to use
  api: "https://api.statsig.com/v1",
  disableAllLogging: false,
  disableLogging: false,
  localMode: false,
  loggingIntervalMS: 10000,     // Flush logs every 10 seconds
  rulesUpdatedCallback: (rules, time) => {
    console.log(`Rules updated at ${time}`);
  }
};

const statsig = new Statsig("secret-server-key", options);
await statsig.initialize();
```
